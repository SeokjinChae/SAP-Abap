FUNCTION ZB23_GET_AIRLINE_INFO.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_CARRID) TYPE  S_CARR_ID
*"  EXPORTING
*"     REFERENCE(ES_AIRLINE) TYPE  ZSSCARR_B23
*"  TABLES
*"      ET_SPFLI TYPE  ZTSPFLI_B23
*"  EXCEPTIONS
*"      NO_DATA
*"----------------------------------------------------------------------

*       Global data declarations

  SELECT SINGLE * INTO CORRESPONDING FIELDS OF es_airline
           FROM scarr
          WHERE carrid = iv_carrid.

  SELECT * INTO CORRESPONDING FIELDS OF TABLE et_spfli
           FROM spfli
          WHERE carrid = iv_carrid.


  IF ( es_airline IS INITIAL AND et_spfli IS INITIAL ) .
    RAISE no_data.
  ENDIF.



ENDFUNCTION.
