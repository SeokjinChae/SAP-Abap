REPORT zb23_select_single.

TYPES ts_SPFLI TYPE bc400_s_connection.

DATA gs_spfli TYPE ts_SPFLI.

SELECT SINGLE *
         INTO CORRESPONDING FIELDS OF gs_spfli
         FROM spfli
        WHERE carrid = 'AA'   " <, >, <>, <=, >= ....
           AND connid = '0017'. "GT, LT, GE, LE, NE ...   "OR AND

  WRITE: / gs_spfli-carrid,
  gs_spfli-connid,
  gs_spfli-cityfrom,
  gs_spfli-airpfrom,
  gs_spfli-cityto,
  gs_spfli-airpto,
  gs_spfli-fltime,
  gs_spfli-deptime,
  gs_spfli-arrtime.
