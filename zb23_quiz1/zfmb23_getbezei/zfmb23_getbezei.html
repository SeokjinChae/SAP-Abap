FUNCTION ZFMB23_GETBEZEI.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_AUART) TYPE  AUART
*"  EXPORTING
*"     REFERENCE(EV_BEZEI) TYPE  TVAKT_BEZEI
*"----------------------------------------------------------------------

*       Global data declarations

SELECT SINGLE bezei
  INTO ev_bezei
  FROM tvakt
  WHERE auart = IV_AUART
  AND spras = sy-langu.



ENDFUNCTION.
