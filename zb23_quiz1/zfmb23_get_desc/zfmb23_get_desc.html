FUNCTION zfmb23_get_desc.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_VBTYP) TYPE  VBTYPL
*"  EXPORTING
*"     REFERENCE(EV_DESC) TYPE  C
*"----------------------------------------------------------------------

*       Global data declarations

  CASE iv_vbtyp.
    WHEN 'A'.
      ev_desc = 'Inquiry'.
    WHEN 'B'.
      ev_desc = 'Quotation'.
    WHEN 'C'.
      ev_desc = 'Order'.
  ENDCASE.




  ENDFUNCTION.
