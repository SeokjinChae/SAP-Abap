DATA : BEGIN OF gs_vbak.
         INCLUDE STRUCTURE vbak.
DATA :   BEZEI_txt TYPE tvakt_bezei,
         desc      TYPE c LENGTH 10,
       END OF gs_vbak.

SELECT SINGLE *
  FROM vbak
  INTO CORRESPONDING FIELDS OF gs_vbak.

CALL FUNCTION 'ZFMB23_GETBEZEI'
  EXPORTING
    iv_auart = gs_vbak-auart
  IMPORTING
    ev_bezei = gs_vbak-BEZEI_txt.


CALL FUNCTION 'ZFMB23_GET_DESC'
  EXPORTING
    iv_vbtyp = gs_vbak-vbtyp
  IMPORTING
    ev_desc  = gs_vbak-desc.
